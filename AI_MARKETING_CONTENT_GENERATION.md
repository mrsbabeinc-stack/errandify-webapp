## PART 6: DATABASE & TRACKING

```sql
CREATE TABLE ai_generated_content (
  id SERIAL PRIMARY KEY,
  content_type VARCHAR(50), -- blog, email, social, ad, etc
  platform VARCHAR(50), -- instagram, facebook, google, blog, etc
  title VARCHAR(255),
  content TEXT,
  generated_by_ai BOOLEAN DEFAULT true,
  approved_by_human BOOLEAN DEFAULT false,
  approved_at TIMESTAMP,
  approved_by_admin_id INTEGER,
  published_at TIMESTAMP,
  
  -- Performance metrics
  impressions INTEGER DEFAULT 0,
  clicks INTEGER DEFAULT 0,
  conversions INTEGER DEFAULT 0,
  engagement_rate DECIMAL(5,2),
  ctr DECIMAL(5,2), -- click-through rate
  conversion_rate DECIMAL(5,2),
  revenue DECIMAL(10,2),
  cost DECIMAL(10,2),
  roi DECIMAL(8,2),
  
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP,
  
  INDEX idx_published (published_at),
  INDEX idx_performance (roi DESC)
);

CREATE TABLE marketing_budget_allocation (
  id SERIAL PRIMARY KEY,
  month DATE,
  channel VARCHAR(50), -- google_ads, facebook, email, blog, etc
  budget_allocated DECIMAL(10,2),
  budget_spent DECIMAL(10,2),
  revenue_generated DECIMAL(10,2),
  roi DECIMAL(8,2),
  recommendation VARCHAR(50), -- scale_up, optimize, maintain, scale_down
  
  INDEX idx_month_channel (month, channel)
);
```
