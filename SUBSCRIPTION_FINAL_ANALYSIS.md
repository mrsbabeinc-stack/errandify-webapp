```sql
-- New table
CREATE TABLE company_subscriptions (
  id INT PRIMARY KEY,
  company_id INT,
  tier ENUM('silver', 'gold', 'platinum'),
  status ENUM('active', 'canceled', 'paused'),
  billing_date DATE,
  next_billing_date DATE,
  stripe_subscription_id VARCHAR(255)
);

-- New table
CREATE TABLE subscription_ad_credits (
  id INT PRIMARY KEY,
  company_id INT,
  month VARCHAR(20),
  allocated_amount INT,
  used_amount INT,
  expires_at DATE
);
```

**2. Backend APIs** (Moderate)
- GET /subscription/status
- POST /subscription/upgrade
- POST /subscription/downgrade
- GET /subscription/billing-history
- POST /ads/apply-credit

**3. Commission Logic Update** (Simple)
```javascript
function getCommissionRate(companyId) {
  const subscription = db.subscriptions.findOne({ company_id: companyId });
  
  if (!subscription || subscription.status !== 'active') {
    return 0.20; // Free users pay 20%
  }
  
  return {
    'silver': 0.18,
    'gold': 0.17,
    'platinum': 0.16
  }[subscription.tier];
}
```

**4. Frontend Dashboard** (Easy)
- Show subscription status
- Show ad credit balance
- Show team members used/limit
- Upgrade/downgrade buttons
- Billing history

**5. Monthly Ad Credit Reset** (Simple Cron)
```javascript
cron.schedule('0 0 1 * *', async () => {
  // Reset ad credits on 1st of month
  const allSubscriptions = await db.subscriptions.find({ status: 'active' });
  for (const sub of allSubscriptions) {
    const creditAmount = {
      'silver': 50,
      'gold': 200,
      'platinum': 500
    }[sub.tier];
    
    await db.subscription_ad_credits.insert({
      company_id: sub.company_id,
      month: getCurrentMonth(),
      allocated_amount: creditAmount,
      used_amount: 0
    });
  }
});
```
